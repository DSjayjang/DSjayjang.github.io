# ggplot 기본 구조
df $>$
    ggplot(aes(x = x축 인자, y = y축 인자)) +
    타겟 plot() + # geom_point() / geom_line() / geom_bar() 등
    labs(title = "제목", x = x축 이름, y = y축 이름) +
    ...



# boxplot
```
df %>%
    ggplot(aes(colnames)) +
    geom_boxplot()
```

### 카테고리별 boxplot 예시
aes 안에 column 여러개 적어주기
```
df %>%
    ggplot(aes(col1, col2)) +
    geom_boxplot()
```

### 박스플롯 색깔 채우기
aes안에 fill 이용
```
df %>%
    ggplot(aes(col1, col2, fill = 값)) +
    geom_boxplot()
```

## 데이터 개수별 박스 크기 조절
varwidth = T/F
```
df %>%
    ggplot() +
    geom_boxplot(varwidth = T)
```

### outlier에 커스텀
outlier.colour = 색깔
outlier.shape = 모양
outlier.size = 숫자
```
df %>%
    ggplot(aes(colnames)) +
    geom_boxplot(outlier.colour = "red",
                outlier.shape = "square",
                outlier.size = 숫자)
```

### geom_jitter()
박스plot에 각 데이터 표시하기
```
df %>%
    ggplot(aes(colnames)) +
    geom_boxplot() +
    geom_jitter()
```

### 수평선 및 수직선 넣어주기
```
df %>%
    ggplot(aes(colnames)) +
    geom_boxplot() + 
    geom_hline(yintercept = 숫자) +  # 수평선
    geom_vline(xintercept = 숫자) +  # 수직선
```

# facet_wrap
plot을 여러개로 나누어줌. (각 변수에 따라 분할 가능)
facet_wrap(~ 나눌 변수)
ncol = 각 plot이 배치될 열의 수
strip.position = 'left' / 'right' 등... : parameter의 위치
scale : 각 plot마다 축의 범위를 다르게 보여줌
    'free_x' / 'free_y' / 'free'
```
df %>%
    ggplot(aes(colnames)) +
    geom_point() + 
    facet_wrap(~ 컬럼명)
```
# facet_grid
~로 각각 변수를 x, y로 나누어줌
x, y값 다 입력해줘야 함.



# geom_smooth
추세선을 그려줌
method : lm(선형회귀), glm(일반화 선형모델), loess(지역다항회귀) 등
formula : 추세선을 적합시킬때 사용할 모델식 (ex) y~x
se : 표준오차를 표시할지 여부 (95%의 신뢰구간)

geom_smooth(method = "", se = "T/F")
추세선 색깔 및 크기 조정
geom_smooth(method = "lm", col = "색깔", linewidth = 크기)
```
df %>%
    ggplot(aes(colnames)) +
    geom_point() + 
    facet_wrap(~ 컬럼명) +
    geom_smooth(method = "lm", se = "F")
```

## 막대 커스텀
fill : 채울 색깔
col : 테두리 색깔
alpha : 막대 투명도 (겹칠때)
position = 'identity' : 히스토그램의 막대를 위로 쌓지 않고 겹치게 함 (ggplot(aes(fill = ~))랑 함께 써야...)
```
df %>%
    ggplot(aes(x = x값)) +
    geom_histogram(bins = 막대개수,
                    fill = 색깔,
                    col = 색깔,
                    alpha = 숫자)
```
